 $ curl -o- https://raw.githubusercontent.com/creationix/nvm/v0.33.8/install.sh | bash

  $ npm install -g cmmc-telegraf@1.0.3

  $ mkdir ~/tick; cd ~/tick
  $ cmmc-telegraf --init
  $ cmmc-telegraf --create [appname]
  $ cmmc-telegraf --debug [appname]
  $ cmmc-telegraf --run [appname]
  $ cmmc-telegraf --warp
